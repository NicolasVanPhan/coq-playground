Module Mynat.

Inductive mynat : Set :=
| O : mynat
| S : mynat → mynat
.

Fixpoint add (n m : mynat) : mynat :=
  match n with
  | O ⇒ m
  | S n' ⇒ S (add n' m)
  end.

Now let's prove some basics on additions

Theorem thm1 : ∀ n : mynat, add n O = n.
Proof.
  induction n.
  - reflexivity.
  - simpl. rewrite IHn. reflexivity.
Qed.

Theorem thm2 : ∀ n : mynat, add O n = n.
Proof.
  induction n.
  - constructor.
  - simpl. reflexivity.
Qed.

Lemma thm3 : ∀ n m : mynat, add n (S m) = S (add n m).
Proof.
  induction n.
  - simpl. reflexivity.
  - intro m. simpl. rewrite IHn. reflexivity.
Qed.

Theorem thm4 : ∀ n m : mynat, add n m = add m n.
Proof.
  induction n.
  -
    intro m. simpl. rewrite (thm1 m). reflexivity.
  -
    intro m.
    simpl.
    rewrite → thm3.
    rewrite IHn.
    reflexivity.
Qed.

My first intuitiion for mult is : 
  4 * 3 = 4 + 4 + 4
        = 4 + (4 * 2)
        = 4 + (4 + (4 * 1))
        = 4 + (4 + (4 + (4 * 0)))
        = 4 + (4 + (4 + 0))
        = 4 + 4 + 4 + 0
        = 4 + 4 + 4
The recursive pattern here is popping out a '4' and decrement the multiplier. n × m = n + (n × (m-1))

Fixpoint mult_v1 (n m : mynat) : mynat :=
  match m with
  | O ⇒ O
  | S m' ⇒ add n (mult_v1 n m')
  end.

End Mynat.
